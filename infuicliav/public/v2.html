# Start the Vault (pick one)
docker run inflectiv/vault
# or
pip install inflectiv-vault

# Store a credential (encrypted at rest)
inflectiv vault store-credential \
  --name "openai_key" \
  --value "sk-xxxxxxxxxxxxxxxx"

# Create a vault from a Knowledge Vault dataset
inflectiv vault create --from defi/governance --mode live
# mode=snapshot (free, one-time seed)
# mode=live     (synced, costs $INAI per cycle)

# Verify
inflectiv vault status
# Vault: running
# Source: defi/governance (live)
# Memory entries: 2,847 (seeded)
# Credentials: 1 (encrypted)
# Network: connected (6,700+ vaults)
from inflectiv import Vault

# Connect to local vault
vault = Vault()

# Store a memory
vault.remember(
    content="Aave V4 proposal #127 passed with 67%",
    tags=["defi", "governance", "aave"]
)

# Query vault + network in one call
result = await vault.query(
    "What happened with Aave governance?",
    sources=["local", "network"]
)

# Create a vault from a Knowledge Vault
vault.create_from("defi/governance", mode="live")
# mode="snapshot" → free, one-time seed
# mode="live"     → synced, costs $INAI

# Get a credential (never exposed to LLM)
key = vault.get_credential("openai_key")
import { Vault } from 'inflectiv-vault'

// Connect to local vault
const vault = new Vault()

// Store a memory
await vault.remember({
  content: 'Aave V4 proposal #127 passed with 67%',
  tags: ['defi', 'governance', 'aave']
})

// Query vault + network
const result = await vault.query({
  query: 'What happened with Aave governance?',
  sources: ['local', 'network']
})

// Create a vault from a Knowledge Vault

// mode: 'snapshot' → free, one-time seed
